349  template<unsigned int N1,unsigned int N2>
350  rc522::status calcCRC(std::array<uint8_t, N1> & dataIn, std::array<uint8_t, N2> & dataOut, bool containsCrc = false) {
351  static_assert(N1 < 64, "The dataIn array is too big, no space in FIFO.");
352  static_assert(N2 > 1, "The dataOut array is too small, no space for CRC results.");
353  std::array<uint8_t, (N1-2)> smallerIn = {};
354  if (containsCrc){
355  std::copy(begin(dataIn),std::prev(end(dataIn),2), begin(smallerIn));
356  }
357 
358  writeReg(rc522::registers::CommandReg, rc522::regCommands::Idle); // Command Idle
359  writeReg(rc522::registers::DivIrqReg, 0x04); // DivIrqReg 0x04 to clear the CRCIrq bit
360  writeReg(rc522::registers::FIFOLevelReg, 0x80); // FIFOLevelReg 0x80 to empty the buffer
361  if (containsCrc){
362  writeReg(rc522::registers::FIFODataReg, smallerIn);
363  } else {
364  writeReg(rc522::registers::FIFODataReg, dataIn); // FIFODataReg, data to write the data to the buffer
365  }
366  writeReg(rc522::registers::CommandReg, rc522::regCommands::CalcCRC); // CommandReg calcCRC to start the calc
367 
368  for (uint16_t i = 5000; i > 0; i--) {
369  uint8_t n = readReg(rc522::registers::DivIrqReg);
370  if (n & 0x04) { // Calc is done, the CRCIrq bit is set
371  writeReg(rc522::registers::CommandReg, rc522::regCommands::Idle); // Stop the calc for new data
372  dataOut[(dataOut.size()-2)] = readReg(rc522::configuration::CRCResultReg_Low); // put the results in the last 2 bytes of dataOut
373  dataOut[(dataOut.size()-1)] = readReg(rc522::configuration::CRCResultReg_High);
374  return rc522::status::SUCCESS;
375  }
376  }
377  return rc522::status::TIMEOUTDEV; // Some time passed communications with the rc522 might be down
378  }

391  template<unsigned int N1, unsigned int N2>
392  rc522::status cardCommunication(
393  std::array<uint8_t,N1> & bufSend,
394  std::array<uint8_t,N2> & bufReceive,
395  rc522::regCommands comData,
396  uint8_t bitFraming = 0,
397  uint8_t waitIrq = 0,
398  bool checkCrc = false
399  ){
400  static_assert(N1 < 64, "bufSend array too big for FIFO register.");
401  writeReg(rc522::registers::CommandReg, rc522::regCommands::Idle);
402  writeReg(rc522::registers::ComIrqReg, 0x7F);
403  writeReg(rc522::registers::FIFOLevelReg, 0x80);
404  for (auto byte : bufSend){
405  writeReg(rc522::registers::FIFODataReg, byte);
406  }
407  writeReg(rc522::registers::BitFramingReg, bitFraming);
408  writeReg(rc522::registers::CommandReg, comData);
409 
410  if (comData == rc522::regCommands::Transceive){
411  setRegBitMask(rc522::registers::BitFramingReg, 0x80); // start communication transceive
412  }
413 
414  // TODO opdelen in kleinere functions ivm leesbaarheid , groeperen low prio
415  // Wait for a response or time out
416  uint16_t i;
417  for (i = 2000; i > 0; i--) {
418  uint8_t regIrq = readReg(rc522::registers::ComIrqReg);
419  if (regIrq & waitIrq) {break;}
420  if (regIrq & 0x01) {return rc522::status::TIMEOUTCOM;}
421  }
422  if (i == 0) {return rc522::status::TIMEOUTDEV;}
423 
424  uint8_t errorVal = readReg(rc522::registers::ErrorReg); // Check for errors except collision
425  if (errorVal & 0x13){return rc522::status::ERROR;}
426  // Get returned data
427  uint8_t _validBits = 0;
428  uint8_t levelFIFO = readReg(rc522::registers::FIFOLevelReg);
429 
430  if (levelFIFO > bufReceive.size()) {return rc522::status::NO_SPACE;}
431  for (uint8_t i = 0; i < levelFIFO; i++){
432  bufReceive[i] = readReg(rc522::registers::FIFODataReg);
433  }
434  // Check the number of valid bits in the last received byte
435  _validBits = readReg(rc522::registers::ControlReg) & 0x07;
436  if (bitFraming) {bitFraming = _validBits;}
437 
438  // Validate CRC if checkCrc true
439  if (bufReceive.size() && checkCrc) {
440  if (bufReceive.size() < 2 || _validBits != 0) {return rc522::status::CRC_ERROR;} // We need the CRC value and the 8 bits of the last byte
441  // do the calculation and save it
442  std::array<uint8_t,2> calcBuf = {};
443  rc522::status caclStatus = calcCRC(bufReceive, calcBuf,true);
444  if (caclStatus != rc522::status::SUCCESS) {
445  return caclStatus;
446  }
447  if ((calcBuf[0] != bufReceive[bufReceive.size() - 2] ) || (calcBuf[1] != bufReceive[bufReceive.size() - 1])) {
448  return rc522::status::CRC_ERROR;
449  }
450  }
451  return rc522::status::SUCCESS;
452  }
